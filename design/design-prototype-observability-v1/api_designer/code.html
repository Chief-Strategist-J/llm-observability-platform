import obsidian_lens as ol
from fastapi import FastAPI, Request

app = FastAPI()

@app.post("/v1/predict")
async def generate_completion(req: Request):
    # Initialize Obsidian Trace context
    trace = ol.trace(name="auth_completion")

    payload = await req.json()
    prompt = payload.get("prompt")

    # Execute LLM call with observability
    response = await ol.llm_call(
        model="gpt-4-obsidian",
        messages=[{"role": "user", "content": prompt}],
        trace=trace
    )

    return {"id": trace.id, "status": "success", "data": response}
